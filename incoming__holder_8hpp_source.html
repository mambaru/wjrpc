289 template<typename J>
290 std::unique_ptr<typename J::target> incoming_holder::get_result( json_error* e) const
291 {
292  if ( !this->has_result() )
293  return nullptr;
294  if ( 'n'==*_incoming.result.first)
295  return nullptr; // is null
296  auto result = std::make_unique<typename J::target>();
297 
298  typename J::serializer()(*result, _incoming.result.first, _incoming.result.second, e);
299 
300  if ( e && *e)
301  return nullptr;
302 
303  return result;
304 }

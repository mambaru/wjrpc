125  void perform_data(data_ptr d, output_handler_t h)
126  {
127  using namespace std::placeholders;
128 
129  while ( d != nullptr )
130  {
131  incoming_holder iholder(std::move(d));
132  ::wjson::json_error e;
133  d = iholder.parse(&e);
134  if ( !e && iholder )
135  {
136  this->invoke( std::move(iholder), [h](outgoing_holder oholder)
137  {
138  h(oholder.detach());
139  });
140  }
141  else
142  {
143  aux::send_error_raw( std::move(iholder), std::make_unique<parse_error>(), h );
144  }
145  }
